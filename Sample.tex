\subsection{微分積分}
\prob{%
  次の極限値を求めよ。
  ただし，\dm{a>0,b>0}とする。
  \begin{gather}
    \lim_{x\to 0}\left(\frac{a^x+b^x}{2}\right)^{1/x}
  \end{gather}
}
\begin{ans*}
  与えられた極限に対数をとった次の極限は
  \begin{align}
    \lim_{x\to 0} \frac{\log \cfrac{a^x + b^x}{2}}{x}
    &= \lim_{x\to 0} \frac{\cfrac{1}{2}\;(a^x\log a+b^x\log b)}{\cfrac{a^x + b^x}{2}} \quad(\text{L'H\^{o}pitalの定理})\\
    &= \frac{\log a + \log b}{2} = \log \sqrt{ab}
  \end{align}
  と得られるので，
  \begin{align}
    \lim_{x\to 0}\left(\frac{a^x+b^x}{2}\right)^{1/x}
     = \sqrt{ab}
  \end{align}
\end{ans*}
